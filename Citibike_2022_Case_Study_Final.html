# Data Processing Overview import pandas as pd import numpy as np import matplotlib.pyplot as plt import seaborn as sns import folium # Load and clean 2022 Citibike data df = pd.read_csv('2022-citibike-tripdata.csv') df['started_at'] = pd.to_datetime(df['started_at']) df['ended_at'] = pd.to_datetime(df['ended_at']) df['trip_duration'] = (df['ended_at'] - df['started_at']).dt.total_seconds() / 60 # Remove outliers and clean data df = df[(df['trip_duration'] > 1) & (df['trip_duration'] < 120)]
